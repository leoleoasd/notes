    \1[6.] 求LASTOP \par

\begin{algorithm}[H]
    \caption{求Lastop}
    \begin{algorithmic}[1]
        \label{algo:1}
        \Procedure{LASTOP}{}
            \For{$A \to \dots a \in P$}
                \State $\lastop(A) \gets \lastop(A) \cup \set{a}$
            \EndFor
            \For{$A \to \dots aB \in P$}
                \State $\lastop(A) \gets \lastop(A) \cup \set{a}$
            \EndFor
            \For{$A \to \dots B \in P$}
                \State $\lastop(A) \gets \lastop(A) \cup \lastop(B)$
            \EndFor
        \EndProcedure
    \end{algorithmic}
\end{algorithm}
